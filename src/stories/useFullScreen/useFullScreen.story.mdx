```js
import { useRef, useState } from 'react'
import { useFullScreen } from 'wrench-hooks'

cconst FullScreen = () => {
  const ref = useRef(null)
  const [show, setShow] = useState(false);
  const isFullscreen = useFullscreen(ref, show, () => setShow(!show));

  return (
    <div ref={ref} style={{ backgroundColor: 'white' }}>
      <div>{isFullscreen ? 'Fullscreen' : 'Not fullscreen'}</div>
      <button onClick={() => setShow(!show)}>Toggle</button>
      <video src="http://clips.vorwaerts-gmbh.de/big_buck_bunny.mp4" autoPlay />
    </div>
  );
}

export default FullScreen
```
